                function startPricingTimer() {
                    const timerElements = document.querySelectorAll('#pricingTimer span');
                    let timeLeft = 2 * 3600 + 18 * 60 + 47; // 02:18:47 in seconds, initial value
                    
                    const timer = setInterval(() => {
                        timeLeft--;
                        
                        if (timeLeft < 0) {
                            clearInterval(timer);
                            // Optionally hide or change the timer display when expired
                            if (timerElements.length > 0) {
                                timerElements.forEach(el => el.textContent = '00');
                            }
                            document.getElementById('pricingTimer').textContent = 'Expired';
                            return;
                        }
                        
                        const hours = Math.floor(timeLeft / 3600);
                        const minutes = Math.floor((timeLeft % 3600) / 60);
                        const seconds = timeLeft % 60;
                        
                        if (timerElements.length >= 3) {
                            timerElements[0].textContent = hours.toString().padStart(2, '0');
                            timerElements[1].textContent = minutes.toString().padStart(2, '0');
                            timerElements[2].textContent = seconds.toString().padStart(2, '0');
                        }
                    }, 1000);
                }

                // ROI Calculator Logic
                function setupRoiCalculator() {
                    const avgCourseFeeInput = document.getElementById('avgCourseFee');
                    const conversionRateInput = document.getElementById('conversionRate');
                    const costPerLeadInput = document.getElementById('costPerLead');
                    const numLeadsInput = document.getElementById('numLeads');
                    const calculateRoiBtn = document.getElementById('calculateRoiBtn');

                    const displayEnrollments = document.getElementById('displayEnrollments');
                    const displayRevenue = document.getElementById('displayRevenue');
                    const displayAdSpend = document.getElementById('displayAdSpend');
                    const displayROI = document.getElementById('displayROI');

                    function calculateROI() {
                        const avgCourseFee = parseFloat(avgCourseFeeInput.value);
                        const conversionRate = parseFloat(conversionRateInput.value) / 100; // Convert percentage to decimal
                        const costPerLead = parseFloat(costPerLeadInput.value);
                        const numLeads = parseFloat(numLeadsInput.value);

                        if (isNaN(avgCourseFee) || isNaN(conversionRate) || isNaN(costPerLead) || isNaN(numLeads) ||
                            avgCourseFee < 0 || conversionRate < 0 || conversionRate > 1 || costPerLead < 0 || numLeads < 0) {
                            // console.error('Please enter valid positive numbers for all fields.');
                            // Instead of alert, we can update a UI message or skip calculation
                            return;
                        }

                        // Calculate Enrollments
                        const expectedEnrollments = Math.floor(numLeads * conversionRate);

                        // Calculate Estimated Revenue
                        const estimatedRevenue = expectedEnrollments * avgCourseFee;

                        // Calculate Estimated Ad Spend
                        const estimatedAdSpend = numLeads * costPerLead;

                        // Calculate ROI
                        let roi = 0;
                        if (estimatedAdSpend > 0) {
                            roi = ((estimatedRevenue - estimatedAdSpend) / estimatedAdSpend) * 100;
                        } else if (estimatedRevenue > 0) {
                            roi = Infinity; // If no ad spend but revenue, ROI is infinite
                        }

                        // Display results
                        displayEnrollments.textContent = expectedEnrollments;
                        displayRevenue.textContent = `$${estimatedRevenue.toFixed(2)}`;
                        displayAdSpend.textContent = `$${estimatedAdSpend.toFixed(2)}`;
                        displayROI.textContent = roi === Infinity ? 'Infinite!' : `${roi.toFixed(2)}%`;
                    }

                    // Initial calculation on load
                    calculateROI();

                    // Add event listener to the button
                    calculateRoiBtn.addEventListener('click', calculateROI);

                    // Add event listeners for input changes to update automatically
                    avgCourseFeeInput.addEventListener('input', calculateROI);
                    conversionRateInput.addEventListener('input', calculateROI);
                    costPerLeadInput.addEventListener('input', calculateROI);
                    numLeadsInput.addEventListener('input', calculateROI);
                }
